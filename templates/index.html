Convert to grayscale first (classic photobooth look) if request.form.get('color_mode') != 'color': image = image.convert('L').convert('RGB') # Adjust contrast and brightness for vintage look enhancer = ImageEnhance.Contrast(image) image = enhancer.enhance(1.2) enhancer = ImageEnhance.Brightness(image) image = enhancer.enhance(0.9) #
